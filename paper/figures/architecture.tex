%% Two-panel architecture figure for MallaNet (PreAct SE-ResNet).
%% Panel (a): high-level pipeline.
%% Panel (b): unravelled PreAct SE-ResBlock.
%%
%% Design target: rendered_height = pdf_height * (346 / pdf_width) >= 270pt
%% If pdf_width ~ 900pt we need pdf_height ~ 702pt.

\begin{document}
\begin{tikzpicture}[
  font=\small,
  arr/.style={-{Latex[length=2.5mm,width=1.5mm]}, line width=0.6pt},
  darr/.style={-{Latex[length=2mm,width=1.2mm]}, dashed, color=gray!70, line width=0.5pt},
  %% Panel (a) block styles
  pblk/.style={draw, rounded corners=4pt, align=center, inner sep=6pt,
               minimum height=22mm, line width=0.6pt},
  inp/.style ={pblk, fill=white,      minimum width=16mm},
  stm/.style ={pblk, fill=gray!20,    minimum width=26mm},
  stg/.style ={pblk, fill=blue!14,    minimum width=30mm},
  hd/.style  ={pblk, fill=teal!18,    minimum width=22mm},
  fc/.style  ={pblk, fill=orange!18,  minimum width=22mm},
  %% Panel (b) main-path node styles
  op/.style={draw, rounded corners=3pt, align=center, inner sep=6pt,
             minimum height=16mm, minimum width=26mm,
             fill=blue!10, line width=0.6pt},
  %% SE sub-block nodes
  seop/.style={draw, rounded corners=3pt, align=center, inner sep=5pt,
               minimum height=14mm, minimum width=22mm,
               fill=violet!12, line width=0.6pt},
  circ/.style={draw, circle, inner sep=5pt, line width=0.6pt, fill=white, font=\normalsize},
  ann/.style={align=left, font=\footnotesize, text=gray!80},
]

%% =========================================================================
%%  PANEL (a) -- High-level pipeline (horizontal row)
%% =========================================================================

\node[inp] (in)
  {\textbf{Input}\\[2pt]$1{\times}32{\times}32$};

\node[stm, right=9mm of in] (stem)
  {\textbf{Stem}\\[2pt]Conv $3{\times}3$, s1\\BN $\to$ ReLU\\$40{\times}32{\times}32$};

\node[stg, right=9mm of stem] (s1)
  {\textbf{Stage 1}\\[2pt]$2{\times}$ PreAct SE\\no stride\\$40{\times}32{\times}32$\\[2pt]{\small 39\,k params}};

\node[stg, right=9mm of s1] (s2)
  {\textbf{Stage 2}\\[2pt]$2{\times}$ PreAct SE\\stride 2\\$80{\times}16{\times}16$\\[2pt]{\small 154\,k params}};

\node[stg, right=9mm of s2] (s3)
  {\textbf{Stage 3}\\[2pt]$2{\times}$ PreAct SE\\stride 2\\$160{\times}8{\times}8$\\[2pt]{\small 616\,k params}};

\node[hd, right=9mm of s3] (gap)
  {BN $\to$ ReLU\\[2pt]Global\\AvgPool\\$160$-dim};

\node[fc, right=9mm of gap] (clf)
  {Dropout $p{=}0.1$\\[2pt]FC $160{\to}46$\\46 classes};

\foreach \a/\b in {in/stem, stem/s1, s1/s2, s2/s3, s3/gap, gap/clf}
  \draw[arr] (\a.east) -- (\b.west);

\node[font=\small\bfseries, below=5mm of s2] (labA)
  {(a)~Full pipeline};

%% Dividing rule
\draw[gray!50, line width=0.6pt]
  ([yshift=-13mm]in.south west) -- ([yshift=-13mm]clf.south east);

%% =========================================================================
%%  PANEL (b) -- Unravelled PreAct SE-ResBlock
%% Panel label
%% =========================================================================

%% ---- Row 1 (main path, left→right) ----
%%  blkin  →  bn1  →  relu1  →  cv1  →  bn2  →  relu2  →  cv2  →  mul → add → blkout
%%  All nodes on same baseline, Y = -55mm below panel(a)

\node[op, below=55mm of in.south west, anchor=north west, xshift=0mm] (blkin)
  {\textbf{in}\\$C{\times}H{\times}W$};

\node[op, right=12mm of blkin]  (bn1)   {BN};
\node[op, right=12mm of bn1]    (relu1) {ReLU};
\node[op, right=12mm of relu1]  (cv1)   {Conv\\$3{\times}3$};
\node[op, right=12mm of cv1]    (bn2)   {BN};
\node[op, right=12mm of bn2]    (relu2) {ReLU};
\node[op, right=12mm of relu2]  (cv2)   {Conv\\$3{\times}3$};
\node[circ, right=12mm of cv2]  (mul)   {$\otimes$};
\node[circ, right=20mm of mul]  (add)   {$\oplus$};
\node[op, right=12mm of add]    (blkout){\textbf{out}\\$C{\times}H{\times}W$};

%% WHY annotations -- above main path
\node[ann, above=20mm of bn1] (wbn)
  {\textbf{pre-activation:}\\BN before Conv\\improves gradient flow};
\draw[darr] (wbn.south) -- (bn1.north);

\node[ann, above=20mm of cv1] (wcv)
  {\textbf{$3{\times}3$ kernel:}\\captures local\\stroke patterns};
\draw[darr] (wcv.south) -- (cv1.north);

\node[ann, above=20mm of mul] (wmul)
  {\textbf{$\otimes$:}\\channel-wise scale};
\draw[darr] (wmul.south) -- (mul.north);

%% ---- SE sub-block (vertical column, below cv2) ----
%% Chain goes: cv2 → GAP → FC(C/8) → ReLU → FC(C) → σ → (up to mul)
%% Laid out vertically going DOWN from cv2

\node[seop, below=32mm of cv2]    (se_gap) {GAP};
\node[seop, below=12mm of se_gap] (se_fc1) {FC $C/8$};
\node[seop, below=12mm of se_fc1] (se_r)   {ReLU};
\node[seop, below=12mm of se_r]   (se_fc2) {FC $C$};
\node[seop, below=12mm of se_fc2] (se_sig) {$\sigma$};

%% SE arrows (top-down)
\draw[arr] (cv2.south) -- (se_gap.north);
\draw[arr] (se_gap.south) -- (se_fc1.north);
\draw[arr] (se_fc1.south) -- (se_r.north);
\draw[arr] (se_r.south) -- (se_fc2.north);
\draw[arr] (se_fc2.south) -- (se_sig.north);

%% SE sigma output goes RIGHT then UP to mul.south
\draw[arr] (se_sig.east) -- ++(18mm,0) |- (mul.south);

%% SE annotation (to the left of the SE column)
\node[ann, left=10mm of se_r] (wse)
  {\textbf{SE:} recalibrate\\channel weights ($r{=}8$)};
\draw[darr] (wse.east) -- (se_r.west);

%% ---- Main-path arrows ----
\draw[arr] (blkin.east) -- (bn1.west);
\draw[arr] (bn1.east)   -- (relu1.west);
\draw[arr] (relu1.east) -- (cv1.west);
\draw[arr] (cv1.east)   -- (bn2.west);
\draw[arr] (bn2.east)   -- (relu2.west);
\draw[arr] (relu2.east) -- (cv2.west);
\draw[arr] (cv2.east)   -- (mul.west);
\draw[arr] (mul.east)   -- (add.west);
\draw[arr] (add.east)   -- (blkout.west);

%% ---- Skip path (goes below the SE column) ----
\coordinate (skip_drop)  at ([yshift=-28mm]se_sig.south);
\coordinate (skip_right) at (skip_drop -| add.south);

\draw[arr, line width=0.8pt]
  (blkin.south) -- (blkin.south |- skip_drop)
  -- node[midway, below, font=\footnotesize, text=gray!80, align=center]
       {identity (or $1{\times}1$ Conv $s{=}2$ when downsampling)}
  (skip_right)
  -- (add.south);

%% ---- Background panel box ----
\begin{scope}[on background layer]
  \node[draw=gray!40, rounded corners=5pt, fill=gray!5, line width=0.5pt,
        fit=(blkin)(blkout)(skip_drop)(skip_right)(wbn)(wcv)(wmul)(wse)(se_sig),
        inner sep=10pt] (panelB) {};
\end{scope}

\node[font=\small\bfseries, below=3mm of panelB.south]
  {(b)~PreAct SE-ResBlock --- unravelled};

\end{tikzpicture}
\end{document}
